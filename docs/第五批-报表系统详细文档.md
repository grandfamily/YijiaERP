# 第五批-报表系统详细文档

## 📊 4. 报表系统 (`reports/`)

### 4.1 模块概述
报表系统提供业务数据分析、可视化图表、自定义报表和定时报表功能。

// src/types/reports.ts
export interface Report {
  id: string;
  name: string;
  description: string;
  type: ReportType;
  category: ReportCategory;
  templateId?: string;
  
  // 配置信息
  config: ReportConfig;
  filters: ReportFilter[];
  visualization: VisualizationConfig;
  
  // 权限控制
  visibility: ReportVisibility;
  allowedRoles: UserRole[];
  
  // 调度信息
  schedule?: ReportSchedule;
  
  // 状态信息
  status: ReportStatus;
  lastRunAt?: Date;
  nextRunAt?: Date;
  
  // 系统信息
  createdBy: string;
  createdAt: Date;
  updatedAt: Date;
}

export type ReportType = 
  | 'procurement_summary'
  | 'supplier_performance'
  | 'inventory_status'
  | 'financial_summary'
  | 'production_metrics'
  | 'quality_analysis'
  | 'custom';

export type ReportCategory = 'operational' | 'financial' | 'strategic' | 'compliance';
export type ReportStatus = 'active' | 'inactive' | 'draft' | 'error';
export type ReportVisibility = 'public' | 'private' | 'role_based';

export interface ReportConfig {
  dataSource: DataSource;
  dateRange: DateRange;
  groupBy: string[];
  aggregations: Aggregation[];
  sorting: SortConfig[];
  limit?: number;
}

export interface DataSource {
  type: 'procurement' | 'inventory' | 'finance' | 'suppliers' | 'production';
  tables: string[];
  joins?: JoinConfig[];
}

export interface JoinConfig {
  table: string;
  type: 'inner' | 'left' | 'right' | 'full';
  on: string;
}

export interface DateRange {
  start: Date;
  end: Date;
  type: 'fixed' | 'relative';
  relativePeriod?: RelativePeriod;
}

export type RelativePeriod = 
  | 'last_7_days'
  | 'last_30_days'
  | 'last_3_months'
  | 'last_6_months'
  | 'last_year'
  | 'current_month'
  | 'current_quarter'
  | 'current_year';

export interface Aggregation {
  field: string;
  function: AggregationFunction;
  alias?: string;
}

export type AggregationFunction = 'sum' | 'avg' | 'count' | 'min' | 'max' | 'distinct_count';

export interface SortConfig {
  field: string;
  direction: 'asc' | 'desc';
}

export interface ReportFilter {
  id: string;
  field: string;
  operator: FilterOperator;
  value: any;
  label: string;
  type: FilterType;
  required: boolean;
}

export type FilterOperator = 
  | 'equals'
  | 'not_equals'
  | 'contains'
  | 'starts_with'
  | 'ends_with'
  | 'greater_than'
  | 'less_than'
  | 'between'
  | 'in'
  | 'not_in';

export type FilterType = 'text' | 'number' | 'date' | 'select' | 'multi_select' | 'boolean';

export interface VisualizationConfig {
  type: ChartType;
  settings: ChartSettings;
  layout: LayoutConfig;
}

export type ChartType = 
  | 'table'
  | 'bar_chart'
  | 'line_chart'
  | 'pie_chart'
  | 'area_chart'
  | 'scatter_plot'
  | 'heatmap'
  | 'gauge'
  | 'metric_card';

export interface ChartSettings {
  title?: string;
  subtitle?: string;
  xAxis?: AxisConfig;
  yAxis?: AxisConfig;
  legend?: LegendConfig;
  colors?: string[];
  showDataLabels?: boolean;
  showGrid?: boolean;
}

export interface AxisConfig {
  title: string;
  format?: string;
  min?: number;
  max?: number;
}

export interface LegendConfig {
  show: boolean;
  position: 'top' | 'bottom' | 'left' | 'right';
}

export interface LayoutConfig {
  width?: number;
  height?: number;
  margin?: {
    top: number;
    right: number;
    bottom: number;
    left: number;
  };

export interface ReportSchedule {

  frequency: ScheduleFrequency;

### 4.4 ReportDashboard 仪表板组件

```typescript
// src/components/reports/ReportDashboard.tsx
import React, { useState, useEffect } from 'react';
import { toast } from 'react-hot-toast';
import {
  RefreshIcon,
  CogIcon,
  FullscreenIcon
} from '@heroicons/react/24/outline';

import { usePurchaseRequests } from '../../hooks/useProcurement';
import { useSuppliers } from '../../hooks/useSuppliers';
import { useInventory } from '../../hooks/useInventory';
import { formatCurrency, formatDate } from '../../utils/format';

interface ReportDashboardProps {
  config: DashboardConfig | null;
  onConfigUpdate: (config: DashboardConfig) => void;
  onRefresh: () => void;
}

const ReportDashboard: React.FC<ReportDashboardProps> = ({
  config,
  onConfigUpdate,
  onRefresh
}) => {
  const [widgetData, setWidgetData] = useState<Record<string, any>>({});
  const [globalFilters, setGlobalFilters] = useState<Record<string, any>>({});
  const [isLoading, setIsLoading] = useState(true);
  const [isFullscreen, setIsFullscreen] = useState(false);

  // 数据源
  const purchaseRequests = usePurchaseRequests();
  const suppliers = useSuppliers();
  const inventory = useInventory();

  useEffect(() => {
    if (config) {
      loadDashboardData();
    }
  }, [config, globalFilters]);

  const loadDashboardData = async () => {
    if (!config) return;

    setIsLoading(true);
    try {
      const data: Record<string, any> = {};

      for (const widget of config.widgets) {
        const widgetResult = await loadWidgetData(widget);
        data[widget.id] = widgetResult;
      }

      setWidgetData(data);
    } catch (error) {
      console.error('Failed to load dashboard data:', error);
      toast.error('加载仪表板数据失败');
    } finally {
      setIsLoading(false);
    }
  };

  const loadWidgetData = async (widget: DashboardWidget) => {
    switch (widget.dataSource.type) {
      case 'procurement':
        return await loadProcurementData(widget);
      case 'suppliers':
        return await loadSupplierData(widget);
      case 'inventory':
        return await loadInventoryData(widget);
      default:
        return null;
    }
  };

  const loadProcurementData = async (widget: DashboardWidget) => {
    let data = [...purchaseRequests];

    switch (widget.id) {
      case 'widget-1': // 总采购金额
        const totalAmount = data.reduce((sum, req) => sum + (req.totalAmount || 0), 0);
        return {
          value: totalAmount,
          change: { value: 12.5, isPositive: true },
          format: 'currency'
        };

      case 'widget-2': // 月度采购趋势
        const monthlyData = groupByMonth(data);
        return {
          categories: monthlyData.map(d => d.month),
          series: [{
            name: '采购金额',
            data: monthlyData.map(d => d.amount)
          }]
        };

      default:
        return null;
    }
  };

  const loadSupplierData = async (widget: DashboardWidget) => {
    // 供应商数据处理
    return null;
  };

  const loadInventoryData = async (widget: DashboardWidget) => {
    // 库存数据处理
    return null;
  };

  const groupByMonth = (data: any[]) => {
    const months = ['1月', '2月', '3月', '4月', '5月', '6月'];
    return months.map(month => ({ 
      month, 
      amount: Math.random() * 1000000 
    }));
  };

  if (!config) {
    return (
      <div className="text-center py-12">
        <h3 className="mt-2 text-sm font-medium text-gray-900">无仪表板配置</h3>
        <p className="mt-1 text-sm text-gray-500">请联系管理员配置仪表板。</p>
      </div>
    );
  }

  return (
    <div className={`${isFullscreen ? 'fixed inset-0 z-50 bg-white' : ''}`}>
      {/* 仪表板控制栏 */}
      <div className="bg-white border-b border-gray-200 px-4 py-3">
        <div className="flex items-center justify-between">
          <div>
            <h2 className="text-lg font-medium text-gray-900">{config.name}</h2>
            {config.description && (
              <p className="text-sm text-gray-500">{config.description}</p>
            )}
          </div>

          <div className="flex items-center space-x-2">
            <button
              onClick={loadDashboardData}
              disabled={isLoading}
              className="inline-flex items-center px-3 py-2 border border-gray-300 shadow-sm text-sm leading-4 font-medium rounded-md text-gray-700 bg-white hover:bg-gray-50 disabled:opacity-50"
            >
              <RefreshIcon className={`h-4 w-4 mr-2 ${isLoading ? 'animate-spin' : ''}`} />
              刷新
            </button>

            <button
              onClick={() => setIsFullscreen(!isFullscreen)}
              className="inline-flex items-center px-3 py-2 border border-gray-300 shadow-sm text-sm leading-4 font-medium rounded-md text-gray-700 bg-white hover:bg-gray-50"
            >
              <FullscreenIcon className="h-4 w-4 mr-2" />
              {isFullscreen ? '退出全屏' : '全屏'}
            </button>
          </div>
        </div>
      </div>

      {/* 仪表板网格 */}
      <div className="p-4">
        <div 
          className="grid gap-4"
          style={{
            gridTemplateColumns: `repeat(${config.layout.columns}, 1fr)`
          }}
        >
          {config.widgets.map((widget) => (
            <div
              key={widget.id}
              className="bg-white rounded-lg shadow border border-gray-200 p-6"
              style={{
                gridColumn: `${widget.position.x + 1} / span ${widget.size.width}`,
                gridRow: `${widget.position.y + 1} / span ${widget.size.height}`
              }}
            >
              <h3 className="text-lg font-medium text-gray-900 mb-4">
                {widget.title}
              </h3>
              
              {isLoading ? (
                <div className="animate-pulse">
                  <div className="h-20 bg-gray-300 rounded"></div>
                </div>
              ) : (
                <div>
                  {widget.type === 'metric_card' && widgetData[widget.id] && (
                    <div>
                      <div className="text-3xl font-bold text-gray-900">
                        {formatCurrency(widgetData[widget.id].value)}
                      </div>
                      {widgetData[widget.id].change && (
                        <div className={`text-sm ${
                          widgetData[widget.id].change.isPositive 
                            ? 'text-green-600' 
                            : 'text-red-600'
                        }`}>
                          {widgetData[widget.id].change.isPositive ? '+' : '-'}
                          {Math.abs(widgetData[widget.id].change.value).toFixed(1)}%
                        </div>
                      )}
                    </div>
                  )}
                  
                  {widget.type === 'chart' && widgetData[widget.id] && (
                    <div className="h-64">
                      {/* 这里可以集成图表库如 Chart.js 或 Recharts */}
                      <div className="text-center text-gray-500">
                        图表数据加载完成
                      </div>
                    </div>
                  )}
                </div>
              )}
            </div>
          ))}
        </div>
      </div>

      {/* 最后更新时间 */}
      <div className="px-4 py-2 text-xs text-gray-500 text-center border-t border-gray-200">
        最后更新: {new Date().toLocaleString()}
      </div>
    </div>
  );
};

export default ReportDashboard;

// src/components/reports/ReportBuilder.tsx
import React, { useState } from 'react';
import { toast } from 'react-hot-toast';
import {
  DatabaseIcon,
  ChartBarIcon,
  AdjustmentsIcon,
  EyeIcon,
  SaveIcon,
  XIcon,
  CheckIcon
} from '@heroicons/react/24/outline';

import { getCurrentUser } from '../../utils/helpers';

interface ReportBuilderProps {
  onSave: (report: Omit<Report, 'id' | 'createdAt' | 'updatedAt'>) => void;
  onClose?: () => void;
  onPreview: (reportId: string, filters?: Record<string, any>) => void;
  initialReport?: Partial<Report>;
}

const ReportBuilder: React.FC<ReportBuilderProps> = ({
  onSave,
  onClose,
  onPreview,
  initialReport
}) => {
  const [currentStep, setCurrentStep] = useState(0);
  const [reportData, setReportData] = useState<Partial<Report>>({
    name: '',
    description: '',
    type: 'custom',
    category: 'operational',
    config: {
      dataSource: {
        type: 'procurement',
        tables: []
      },
      dateRange: {
        start: new Date(),
        end: new Date(),
        type: 'relative',
        relativePeriod: 'last_30_days'
      },
      groupBy: [],
      aggregations: [],
      sorting: []
    },
    filters: [],
    visualization: {
      type: 'table',
      settings: {},
      layout: {}
    },
    visibility: 'private',
    allowedRoles: [],
    status: 'draft',
    ...initialReport
  });
  const [previewData, setPreviewData] = useState<ReportData | null>(null);
  const [isPreviewLoading, setIsPreviewLoading] = useState(false);

  const steps = [
    { 
      id: 'basic', 
      title: '基本信息', 
      icon: DatabaseIcon,
      description: '设置报表名称、描述和类型'
    },
    { 
      id: 'data', 
      title: '数据源', 
      icon: DatabaseIcon,
      description: '选择数据源和配置查询条件'
    },
    { 
      id: 'filters', 
      title: '筛选器', 
      icon: AdjustmentsIcon,
      description: '添加报表筛选器和参数'
    },
    { 
      id: 'visualization', 
      title: '可视化', 
      icon: ChartBarIcon,
      description: '配置图表类型和样式'
    },
    { 
      id: 'preview', 
      title: '预览', 
      icon: EyeIcon,
      description: '预览报表效果'
    }
  ];

  const handleStepChange = (stepIndex: number) => {
    if (validateCurrentStep()) {
      setCurrentStep(stepIndex);
    }
  };

  const validateCurrentStep = (): boolean => {
    switch (steps[currentStep].id) {
      case 'basic':
        if (!reportData.name?.trim()) {
          toast.error('请输入报表名称');
          return false;
        }
        if (!reportData.type) {
          toast.error('请选择报表类型');
          return false;
        }
        break;

      case 'data':
        if (!reportData.config?.dataSource.tables.length) {
          toast.error('请选择至少一个数据表');
          return false;
        }
        break;

      case 'visualization':
        if (!reportData.visualization?.type) {
          toast.error('请选择可视化类型');
          return false;
        }
        break;
    }
    return true;
  };

  const handlePreview = async () => {
    if (!validateCurrentStep()) return;

    setIsPreviewLoading(true);
    try {
      // 模拟生成预览数据
      const preview = await generatePreviewData(reportData);
      setPreviewData(preview);
    } catch (error) {
      console.error('Failed to generate preview:', error);
      toast.error('预览生成失败');
    } finally {
      setIsPreviewLoading(false);
    }
  };

  const generatePreviewData = async (report: Partial<Report>): Promise<ReportData> => {
    // 模拟报表数据生成
    const mockData: ReportData = {
      columns: [
        { field: 'id', title: 'ID', type: 'string' },
        { field: 'name', title: '名称', type: 'string' },
        { field: 'amount', title: '金额', type: 'number', format: 'currency' },
        { field: 'date', title: '日期', type: 'date' }
      ],
      rows: [
        ['1', '示例数据1', 1000, new Date()],
        ['2', '示例数据2', 2000, new Date()],
        ['3', '示例数据3', 3000, new Date()]
      ],
      totalRows: 3,
      executionTime: 125,
      generatedAt: new Date()
    };

    return mockData;
  };

  const handleSave = () => {
    if (!validateAllSteps()) return;

    const finalReport: Omit<Report, 'id' | 'createdAt' | 'updatedAt'> = {
      name: reportData.name!,
      description: reportData.description || '',
      type: reportData.type!,
      category: reportData.category!,
      config: reportData.config!,
      filters: reportData.filters || [],
      visualization: reportData.visualization!,
      visibility: reportData.visibility!,
      allowedRoles: reportData.allowedRoles || [],
      status: 'active',
      createdBy: getCurrentUser().id
    };

    onSave(finalReport);
  };

  const validateAllSteps = (): boolean => {
    if (!reportData.name?.trim()) {
      toast.error('请输入报表名称');
      setCurrentStep(0);
      return false;
    }

    if (!reportData.config?.dataSource.tables.length) {
      toast.error('请配置数据源');
      setCurrentStep(1);
      return false;
    }

    if (!reportData.visualization?.type) {
      toast.error('请配置可视化类型');
      setCurrentStep(3);
      return false;
    }

    return true;
  };

  const updateReportData = (updates: Partial<Report>) => {
    setReportData(prev => ({
      ...prev,
      ...updates
    }));
  };

  return (
    <div className="fixed inset-0 z-50 overflow-hidden">
      <div className="absolute inset-0 bg-gray-500 bg-opacity-75" onClick={onClose}></div>
      
      <div className="absolute inset-y-0 right-0 max-w-full flex">
        <div className="w-screen max-w-6xl">
          <div className="h-full flex flex-col bg-white shadow-xl">
            {/* 标题栏 */}
            <div className="px-6 py-4 bg-gray-50 border-b border-gray-200">
              <div className="flex items-center justify-between">
                <h2 className="text-lg font-medium text-gray-900">
                  {initialReport ? '编辑报表' : '创建报表'}
                </h2>
                <div className="flex items-center space-x-3">
                  <button
                    onClick={handlePreview}
                    disabled={isPreviewLoading}
                    className="inline-flex items-center px-3 py-2 border border-gray-300 shadow-sm text-sm leading-4 font-medium rounded-md text-gray-700 bg-white hover:bg-gray-50 disabled:opacity-50"
                  >
                    <EyeIcon className="h-4 w-4 mr-2" />
                    预览
                  </button>
                  <button
                    onClick={handleSave}
                    className="inline-flex items-center px-4 py-2 border border-transparent text-sm font-medium rounded-md text-white bg-blue-600 hover:bg-blue-700"
                  >
                    <SaveIcon className="h-4 w-4 mr-2" />
                    保存
                  </button>
                  {onClose && (
                    <button
                      onClick={onClose}
                      className="text-gray-400 hover:text-gray-500"
                    >
                      <XIcon className="h-6 w-6" />
                    </button>
                  )}
                </div>
              </div>
            </div>

            {/* 步骤导航 */}
            <div className="px-6 py-4 border-b border-gray-200">
              <nav className="flex space-x-8">
                {steps.map((step, index) => {
                  const Icon = step.icon;
                  const isActive = index === currentStep;
                  const isCompleted = index < currentStep;
                  
                  return (
                    <button
                      key={step.id}
                      onClick={() => handleStepChange(index)}
                      className={`
                        flex items-center text-sm font-medium
                        ${isActive 
                          ? 'text-blue-600' 
                          : isCompleted 
                            ? 'text-green-600' 
                            : 'text-gray-500 hover:text-gray-700'
                        }
                      `}
                    >
                      <Icon className="h-5 w-5 mr-2" />
                      <span>{step.title}</span>
                      {isCompleted && (
                        <CheckIcon className="h-4 w-4 ml-1 text-green-500" />
                      )}
                    </button>
                  );
                })}
              </nav>
            </div>

            {/* 步骤内容 */}
            <div className="flex-1 overflow-y-auto">
              <div className="px-6 py-6">
                {currentStep === 0 && (
                  <div className="space-y-6">
                    <div>
                      <h3 className="text-lg font-medium text-gray-900">基本信息</h3>
                      <p className="mt-1 text-sm text-gray-500">
                        设置报表的基本信息和类型
                      </p>
                    </div>

                    <div className="grid grid-cols-1 gap-6 sm:grid-cols-2">
                      <div>
                        <label className="block text-sm font-medium text-gray-700">
                          报表名称 *
                        </label>
                        <input
                          type="text"
                          value={reportData.name || ''}
                          onChange={(e) => updateReportData({ name: e.target.value })}
                          className="mt-1 block w-full border-gray-300 rounded-md shadow-sm focus:ring-blue-500 focus:border-blue-500"
                          placeholder="输入报表名称"
                        />
                      </div>

                      <div>
                        <label className="block text-sm font-medium text-gray-700">
                          报表类型 *
                        </label>
                        <select
                          value={reportData.type || ''}
                          onChange={(e) => updateReportData({ type: e.target.value as ReportType })}
                          className="mt-1 block w-full border-gray-300 rounded-md shadow-sm focus:ring-blue-500 focus:border-blue-500"
                        >
                          <option value="">选择报表类型</option>
                          <option value="procurement_summary">采购汇总</option>
                          <option value="supplier_performance">供应商绩效</option>
                          <option value="inventory_status">库存状态</option>
                          <option value="financial_summary">财务汇总</option>
                          <option value="custom">自定义</option>
                        </select>
                      </div>

                      <div className="sm:col-span-2">
                        <label className="block text-sm font-medium text-gray-700">
                          报表描述
                        </label>
                        <textarea
                          value={reportData.description || ''}
                          onChange={(e) => updateReportData({ description: e.target.value })}
                          rows={3}
                          className="mt-1 block w-full border-gray-300 rounded-md shadow-sm focus:ring-blue-500 focus:border-blue-500"
                          placeholder="输入报表描述"
                        />
                      </div>
                    </div>
                  </div>
                )}

                {currentStep === 4 && (
                  <div className="space-y-6">
                    <div>
                      <h3 className="text-lg font-medium text-gray-900">报表预览</h3>
                      <p className="mt-1 text-sm text-gray-500">
                        预览报表的最终效果
                      </p>
                    </div>

                    {isPreviewLoading ? (
                      <div className="animate-pulse">
                        <div className="h-64 bg-gray-300 rounded"></div>
                      </div>
                    ) : previewData ? (
                      <div className="bg-white border border-gray-200 rounded-lg overflow-hidden">
                        <table className="min-w-full divide-y divide-gray-200">
                          <thead className="bg-gray-50">
                            <tr>
                              {previewData.columns.map((column) => (
                                <th
                                  key={column.field}
                                  className="px-6 py-3 text-left text-xs font-medium text-gray-500 uppercase tracking-wider"
                                >
                                  {column.title}
                                </th>
                              ))}
                            </tr>
                          </thead>
                          <tbody className="bg-white divide-y divide-gray-200">
                            {previewData.rows.map((row, index) => (
                              <tr key={index}>
                                {row.map((cell, cellIndex) => (
                                  <td
                                    key={cellIndex}
                                    className="px-6 py-4 whitespace-nowrap text-sm text-gray-900"
                                  >
                                    {cell}
                                  </td>
                                ))}
                              </tr>
                            ))}
                          </tbody>
                        </table>
                      </div>
                    ) : (
                      <div className="text-center py-12">
                        <p className="text-gray-500">点击预览按钮生成报表预览</p>
                      </div>
                    )}
                  </div>
                )}
              </div>
            </div>

            {/* 底部按钮 */}
            <div className="px-6 py-4 bg-gray-50 border-t border-gray-200">
              <div className="flex items-center justify-between">
                <button
                  onClick={() => setCurrentStep(Math.max(0, currentStep - 1))}
                  disabled={currentStep === 0}
                  className="inline-flex items-center px-4 py-2 border border-gray-300 shadow-sm text-sm font-medium rounded-md text-gray-700 bg-white hover:bg-gray-50 disabled:opacity-50"
                >
                  上一步
                </button>

                <span className="text-sm text-gray-500">
                  第 {currentStep + 1} 步，共 {steps.length} 步
                </span>

                <button
                  onClick={() => {
                    if (currentStep === steps.length - 1) {
                      handleSave();
                    } else if (validateCurrentStep()) {
                      setCurrentStep(currentStep + 1);
                    }
                  }}
                  className="inline-flex items-center px-4 py-2 border border-transparent text-sm font-medium rounded-md text-white bg-blue-600 hover:bg-blue-700"
                >
                  {currentStep === steps.length - 1 ? '完成' : '下一步'}
                </button>
              </div>
            </div>
          </div>
        </div>
      </div>
    </div>
  );
};

export default ReportBuilder;
```
